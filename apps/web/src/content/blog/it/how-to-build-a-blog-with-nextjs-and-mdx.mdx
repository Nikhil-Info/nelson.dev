## Come creare un blog

Prima crea un progetto Next.js con il seguente comando:

```bash title='Terminale'
yarn create next-app nextjs-mdx-blog
```

Quindi crea i seguenti file:

- `components/layout.jsx` - Avvolgere tutti i componenti in un contenitore (opzionale, solo per stile)
- `date/blog/*.mdx` - Articoli del blog
- `lib/format-date.js` - Formattare la data come `DD mese YYYY`
- `pages/blog/[slug].jsx` - pagina dell'articolo, utilizzando [route dinamiche](https://nextjs.org/docs/pages/building-your-application/routing/dynamic-routes)

### Come elaborare i file Markdown

Prima `const root{:js}` - directory root, e il metodo `process.cwd(){:js}` restituisce la directory di lavoro corrente del processo Node.js.

```js title='lib/mdx.js'
const root = process.cwd()
```

L'altra variabile `POSTS_PATH{:.entity.name.constant}` per il percorso in cui sono memorizzati i file degli articoli.

```js title='lib/mdx.js'
import path from 'path'

const POSTS_PATH = path.join(root, 'data', 'blog') // [!code highlight]
// Output: A:/nextjs-mdx-blog/data/blog
```

Quindi usa fs per leggere il contenuto di questa directory, cioè tutti i nomi dei file in `data/blog`.

```js title='lib/mdx.js'
import fs from 'fs'

export const allSlugs = fs.readdirSync(POSTS_PATH)
// Output: ['markdown.mdx', 'nextjs.mdx', 'react.mdx']
```

Quindi scrivi una funzione per rimuovere l'estensione del file, che verrà utilizzata in seguito.

```js title='lib/mdx.js'
export const formatSlug = (slug) => slug.replace(/\.mdx$/, '')
/*
 * Esempio: formatSlug('markdown.mdx')
 * Output: 'markdown'
 */
```

Il passo successivo è ottenere il contenuto dell'articolo tramite slug.

```js title='lib/mdx.js'
export const getPostBySlug = async (slug) => {
  const postFilePath = path.join(POSTS_PATH, `${slug}.mdx`)
  // Output: A:/nextjs-mdx-blog/data/blog/slug.mdx
  const source = fs.readFileSync(postFilePath)
  // Restituisce il contenuto del file

  const { content, data } = matter(source)
  /*
   * Esempio:
   *  ---
   *  title: Ciao
   *  slug: home
   *  ---
   *  <h1>Ciao mondo!</h1>
   *
   * Ritorno:
   *  {
   *    content: '<h1>Ciao mondo!</h1>',
   *    data: {
   *      title: 'Ciao',
   *      slug: 'home'
   *    }
   *  }
   */

  const mdxSource = await serialize(content)
  // Inserire il contenuto per la serializzazione (next-mdx-remote) per l'elaborazione

  const frontMatter = {
    ...data,
    slug
  }
  // Inserire anche slug in front matter, che verrà utilizzato in seguito

  return {
    source: mdxSource,
    frontMatter
  }
}
```

Quindi puoi ottenere tutti gli articoli per visualizzarli nella home page.

```js title='lib/mdx.js'
export const getAllPosts = () => {
  const frontMatter = []

  allSlugs.forEach((slug) => {
    const source = fs.readFileSync(path.join(POSTS_PATH, slug), 'utf-8')

    const { data } = matter(source)

    frontMatter.push({
      ...data,
      slug: formatSlug(slug),
      date: new Date(data.date).toISOString()
    })
  })

  return frontMatter.sort((a, b) => dateSortDesc(a.date, b.date))
}

// Ordinato per data in ordine decrescente
const dateSortDesc = (a, b) => {
  if (a > b) return -1
  if (a < b) return 1

  return 0
}
```

### Formattazione della data

```js title='lib/format-date.js'
export const formatDate = (date) =>
  new Date(date).toLocaleDateString('it', {
    year: 'numeric',
    month: 'long',
    day: 'numeric'
  })
/*
 * formatDate('2022-08-21T00:00:00Z')
 * Output: '21 agosto 2022'
 */
```

### Home page

```jsx title='pages/index.jsx'
import { formatDate } from '../lib/format-date'
import { getAllPosts } from '../lib/mdx'

import Link from 'next/link'

export default function Home({ posts }) {
  return (
    <>
      <h1 className='mb-8 text-6xl font-bold'>Blog</h1>
      <hr className='my-8' />
      <ul className='flex flex-col gap-3'>
        {posts.map(({ slug, title, summary, date }) => (
          <li key={slug}>
            <Link href={`/blog/${slug}`}>
              <a className='block rounded-lg border border-solid border-gray-300 p-6 shadow-md'>
                <div className='flex justify-between'>
                  <h2>{title}</h2>
                  <time dateTime={date}>{formatDate(date)}</time>
                </div>
                <p className='mt-4'>{summary}</p>
              </a>
            </Link>
          </li>
        ))}
      </ul>
    </>
  )
}

// Usa getStaticProps per ottenere tutti gli articoli
export const getStaticProps = async () => {
  const posts = getAllPosts()

  return {
    props: {
      posts
    }
  }
}
```

### Pagina dell'articolo

```jsx title='pages/[slug].jsx'
import { formatDate } from '../../lib/format-date'
import { allSlugs, formatSlug, getPostBySlug } from '../../lib/mdx'

import { MDXRemote } from 'next-mdx-remote'

export default function Blog({ post }) {
  const { title, date } = post.frontMatter

  return (
    <div>
      <h1 className='mb-2 text-6xl font-bold'>{title}</h1>
      <time dateTime={date} className='text-lg font-medium'>
        {formatDate(date)}
      </time>
      <hr className='my-8' />
      <article className='prose max-w-none'>
        <MDXRemote {...post.source} />
      </article>
    </div>
  )
}

export const getStaticProps = async ({ params }) => {
  const post = await getPostBySlug(params.slug)

  return {
    props: {
      post
    }
  }
}

export const getStaticPaths = async () => {
  const paths = allSlugs.map((slug) => ({
    params: {
      slug: formatSlug(slug)
    }
  }))
  /*
   * Output paths:
   *   [
   *     { params: { slug: 'markdown' } },
   *     { params: { slug: 'nextjs' } },
   *     { params: { slug: 'react' } }
   *   ]
   */

  return {
    paths,
    fallback: false
  }
}
```

In questo modo, un semplice blog è pronto.
